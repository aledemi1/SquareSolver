Note
Operation logic:
1 determine the type of equation
2 type: 0x^2 + 0x + c = 0 => return 0(numbers of roots)
3 type: 0x^2 + bx + c = 0 (linear equation) => count answer = (-c)/b => checking that the answer is 0 =>return 1(numbers of roots)
4 type: ax^2 + bx + c = 0 => initialization and count the discriminant
4.a if discriminant < 0 => return 0(numbers of roots) due to the lack of real roots
4.b if discriminant > 0 => count roots = (-b +-sqrt(discriminant)) / (2*a) =>checking that the answers is 0 =>return 2(numbers of roots)
4.c if discriminant = 0 => count root = ( -b ) / (2*a) => checking that the answers is 0 => return 1(numbers of roots)
5 if another situation: ERROR!

if (check_that_number_is_zero(b))

}

initialization and count the discriminant
float D = b*b - 4*a*c;
if (D<0)
{
return 0;
}
else if(D>0)
{
*ans1_address = (float) (-b + sqrt(D)) / (2*a);
*ans2_address = (float) (-b - sqrt(D)) / (2*a);
if (check_that_number_is_zero(*ans1_address))
*ans1_address = 0;
if (check_that_number_is_zero(*ans2_address))
*ans2_address = 0;
return 2;
}
else if(check_that_number_is_zero(D))
{
*ans1_address = (-b) / (2*a);
if (check_that_number_is_zero(*ans1_address))
*ans1_address = 0;
return 1;
}
else
{
assert(0);
return 0;
}
}
